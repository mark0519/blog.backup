C结构体

C语言结构体基本知识

C 数组允许定义可存储相同类型数据项的变量，结构是 C 编程中另一种用户自定义的可用的数据类型，它允许存储不同类型的数据项。

结构体定义

为了定义结构，必须使用 struct 语句。struct语句结构如下：

struct tag
{
    member-list;
    member-list;
    member-list;
    member-list;
    ...
} variable-list ;

tag 是结构体标签。

member-list 是标准的变量定义，比如 int i; 或者 float f，或者其他有效的变量定义。

variable-list 结构变量，定义在结构的末尾，最后一个分号之前，可以指定一个或多个结构变量。

例如下面的定义:

struct books
{
    char title[100];
    char author[100];
    char subject[100];
    int book_id;
};

struct books book1;

等价于：

struct books
{
    char title[100];
    char author[100];
    char subject[100];
    int book_id;
} book1;

在一般情况下，tag、member-list、variable-list 这 3 部分至少要出现 2 个。

结构体变量的初始化

和其它类型变量一样，对结构体变量可以在定义时指定初始值。

例如：

struct books
{
    char title[100];
    char author[100];
    char subject[100];
    int book_id;
} book1={"a book on c","mark0519","programming",13456};

或者

struct books
{
    char title[100];
    char author[100];
    char subject[100];
    int book_id;
};

struct books book1={"a book on c","mark0519","programming",13456};

再或者

struct books
{
    char title[100];
    char author[100];
    char subject[100];
    int book_id;
};

struct books book1;

strcpy(book1.title,"a book on c");         //book1.title中的“.”为“成员访问运算符”，用于访问结构的成员
strcpy(book1.author,"mark0519");
strcpy(book1.subject,"programming");
book1.book_id=123456;

访问结构成员

上面的最后一个例子中的 book1.title 中的 “.” 为 “成员访问运算符” ，用于访问结构的成员。

下面的例子用于理解该运算符：

#include <stdio.h>
#include <string.h>
 
struct Books
{
   char  title[50];
   char  author[50];
   char  subject[100];
   int   book_id;
};
 
int main()
{
   struct Books Book1;        /* 声明 Book1，类型为 Books */
   struct Books Book2;        /* 声明 Book2，类型为 Books */
 
   /* Book1 详述 */
   strcpy( Book1.title, "C Programming");
   strcpy( Book1.author, "Nuha Ali"); 
   strcpy( Book1.subject, "C Programming Tutorial");
   Book1.book_id = 6495407;
 
   /* Book2 详述 */
   strcpy( Book2.title, "Telecom Billing");
   strcpy( Book2.author, "Zara Ali");
   strcpy( Book2.subject, "Telecom Billing Tutorial");
   Book2.book_id = 6495700;
 
   /* 输出 Book1 信息 */
   printf( "Book 1 title : %s\n", Book1.title);
   printf( "Book 1 author : %s\n", Book1.author);
   printf( "Book 1 subject : %s\n", Book1.subject);
   printf( "Book 1 book_id : %d\n", Book1.book_id);
 
   /* 输出 Book2 信息 */
   printf( "Book 2 title : %s\n", Book2.title);
   printf( "Book 2 author : %s\n", Book2.author);
   printf( "Book 2 subject : %s\n", Book2.subject);
   printf( "Book 2 book_id : %d\n", Book2.book_id);
 
   return 0;
}

输出结果为：

Book 1 title : C Programming
Book 1 author : Nuha Ali
Book 1 subject : C Programming Tutorial
Book 1 book_id : 6495407
Book 2 title : Telecom Billing
Book 2 author : Zara Ali
Book 2 subject : Telecom Billing Tutorial
Book 2 book_id : 6495700

结构作为函数参数

可以把结构作为函数参数，传参方式与其他类型的变量或指针类似。可以使用上面实例中的方式来访问结构变量：

#include<stdio.h>
#include<string.h>

void printfbook(struct books book);

struct books
{
    char title[100];
    char author[100];
    char subject[100];
    int book_id;
} book1={"learn c","mark_","programming",654321};

int main()
{
    struct books book2;  
    strcpy(book2.title,"a book on c");         
	strcpy(book2.author,"mark0519");
	strcpy(book2.subject,"programming");
	book2.book_id=123456;
    
    printfbook(book1);
    printfbook(book2);
    
    return 0;
}

void printfbook(struct books book)
{
    printf("\ntitle=%s\n",book.title);
    printf("author=%s\n",book.author);
    printf("subject=%s\n",book.subject);
    printf("book_id=%d\n",book.book_id);
}

运行结果为


title=learn c
author=mark_
subject=programming
book_id=654321

title=a book on c
author=mark0519
subject=programming
book_id=123456
