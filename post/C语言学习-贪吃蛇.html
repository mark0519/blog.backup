前置知识

使用子函数改动main函数中变量的方法

基本原理

在main函数中定义一个变量，然后可以将该变量的地址传给子函数，然后在子函数中以指针型式接受变量地址，即可使用*获得变量值，然后对其进行计算，只要保持其地址不变，即可改变main函数中该变量的值。

例子

例如：

#include<stdio.h>
void change(int* m);//改变变量值的子函数
int main()
{
    int a = 1;
    printf("a=%d\n",a);
    change(&a);     //改变变量值的子函数
    printf("a=%d\n",a);
}
void change(int* m)
{
    *m=*m+1;
}

输出结果为：

a=1
a=2

几个函数

Sleep()

函数名：Sleep()

函数原型：void Sleep(DWORD dwMilliseconds); 参数单位为毫秒

功 能：执行挂起一段时间 　

所属头文件：#include <windows.h>

（ 在linux中使用#include <unistd.h> ）

在VC中,Sleep中的第一个英文字符为大写的”S” ,在linux下不要大写，在标准C中是sleep, 不要大写。

Sleep(3000);  //暂停3秒

system(“cls”)

函数名：system()

参数“cls”作用：执行清屏命令

所属头文件 #include <windows.h>

_kbhit()

函数名：_kbhit() (VC++6.0的更早版本为kbhit() ）

函数原型：int kbhit(void);

功能：检查当前是否有键盘输入。

返回值：若有则返回一个非0值，否则返回 0

所属头文件： #include <conio.h>

if(_kbhit())  //如果有输入则打印字符串 mark
{
	printf("mark\n");
}

_getch()

函数名：_getch() (VC++6.0的更早版本为getch() ）

函数原型：int getch(void)

功能：接受一个任意键的输入，不用按回车就返回，也不打印在屏幕上

返回值：键入字符的ASCII值

所在头文件：#include<conio.h>

switch (_getch())    //根据键下的键来改变变量temp的值
{
	case 'w':case 'W': temp = 0; break; 
	case 'a':case 'A': temp = 3; break; 
	case 's':case 'S': temp = 2; break; 
	case 'd':case 'D': temp = 1; break; 
}

贪吃蛇基本思路

地图空的位置用0表示，食物位置用-1来表示，蛇的位置用1、2、3、。。。来表示，
其中1代表蛇头，2代表紧接着蛇头的一节身体，3代表再下一节身体，以此类推。

具体过程：

1.初始化游戏，包括初始化地图，食物的位置，蛇的位置
2.获取玩家输入的控制命令
3.根据输入的控制命令移动蛇的位置（左转、右转还是方向不变），并且返回移动的结果（是吃到食物，还是撞到障碍物，还是其他）
4.分析移动的结果，如果是吃到了食物，分数加一，撞到了障碍物结束游戏。
5.将蛇画到控制台
6.回到2过程

源码

#include<stdio.h>
#include<time.h>
#include<conio.h>
#include<Windows.h>

#define SIZE 15    //地图大小

int begin(void); //开始界面
void init(int* command, int* length, int map[SIZE][SIZE]);  //初始化地图（初始化地图，食物的位置，蛇的位置）
int getcommand(int command);  //获取命令
int move(int command, int* length, int map[SIZE][SIZE],int* sore);  //移动蛇
void draw(int map[SIZE][SIZE],int id);  //画出蛇

int main()  
{
	int map[SIZE][SIZE] = { 0 };   //创建地图
	int command, result, length = 0;   //初始化长度为0
	int s = 1, sore = 0;

	s = begin(); //开始界面

	init(&command, &length, map);   //初始化地图（初始化地图，食物的位置，蛇的位置）

	while (1)
	{
		command = getcommand(command);
		result = move(command, &length, map,&sore);
		if (result == 1)  //失败
			break;
		else
			draw(map,s);  //画出结果
        
		Sleep(100*s);
	}
	printf("\nOVER!\n\n");
	printf("得分：%d\n",sore);
	system("pause");
	return 0;
}
int begin(void)
{
	int id = 1, flag = 0;

	while (1)
	{
		if (_kbhit())
		{
			switch (_getch()) 
			{
			case 'w':id = id - 1; break;
			case 's':id = id + 1; break;
			case 'a':flag = 1;  break;
			case 'd':flag = 1;  break;
			}
		}

		if (id > 3)
			id = 3;
		if (id < 1)
			id = 1;

		printf("\n     请选择难度:\n");
		printf("\n     (使用‘w’和‘s’控制难度，按‘a’或‘d’确定)\n\n");
		if (id == 1) printf("   -->");
		printf("\t困难\n");
		if (id == 2) printf("   -->");
		printf("\t一般\n");
		if (id == 3) printf("   -->");
		printf("\t简单\n");
		printf("\n\n  Power by Mark0519\n");
		Sleep(80);

		system("cls");
		if (flag == 1)
		{
			return id;
			break;
			break;
		}
	}
}
void init(int* command, int* length, int map[SIZE][SIZE])
{
	*command = 2;//初始化蛇移动方向向下
	*length = 3; //初始化蛇的长度为3

	//初始化整条蛇的位置
	map[2][1] = 1;  //蛇头位置
	map[1][1] = 2;
	map[0][1] = 3;  //蛇尾位置

	//初始化食物的位置
	map[3][3] = -1;  //-1代表食物
}
int getcommand(int command)
{
	int temp = -1;

	if (_kbhit())//如果键盘有输入             
	{
        /*
              0
              ↑ 
 		 3 ←     → 1
 		      ↓
              2
		*/
		switch (_getch())  //操控蛇的输入的字符
		{
		case 'w':case 'W': temp = 0; break; //向上为0
		case 'a':case 'A': temp = 3; break; //向左为3
		case 's':case 'S': temp = 2; break; //向下为2
		case 'd':case 'D': temp = 1; break; //向右为1
		}
		if (temp != -1 && abs(command - temp) != 2)  //蛇不能反向
		{
			command = temp;
		}
	}
	
	return command;
}
int move(int command, int* length, int map[SIZE][SIZE],int* sore)
{
	int i, j, head_i, head_j;

	for (i = 0; i < SIZE; i++)
	{
		for (j = 0; j < SIZE; j++)
		{
			if (map[i][j] == *length) //如果是蛇的最后一节身体
			{
				map[i][j] = 0;
			}
			else if (map[i][j] > 1) //如果是蛇中间节的身体
			{
				map[i][j] += 1;
			}
			else if (map[i][j] == 1) //如果是蛇头
			{
				map[i][j] += 1;
				//确定蛇头新的位置head_i和head_j
				switch (command)
				{
				case 0: head_i = i - 1, head_j = j; break;
				case 1: head_i = i, head_j = j + 1; break;
				case 2: head_i = i + 1, head_j = j; break;
				case 3: head_i = i, head_j = j - 1; break;
				}
			}
		}
	}

	//判断移动结果
	if (map[head_i][head_j] == -1) //如果吃到了食物
	{
		*sore = *sore + 1;
		map[head_i][head_j] = 1;
		*length = *length + 1; //蛇的长度加一
		//重新生成食物
		while (1)
		{
            srand(time(NULL)); //设置种子
			i = rand() % SIZE, j = rand() % SIZE;
			if (map[i][j] == 0)
			{
				map[i][j] = -1;
				break;
			}
		}
	}
	else if (map[head_i][head_j] > 0 || head_i < 0 || head_i == SIZE || head_j < 0 || head_j == SIZE)//撞墙或撞自己
	{
		return 1; //失败
	}
	else
	{
		map[head_i][head_j] = 1;
	}
	return 0; //成功
}
void draw(int map[SIZE][SIZE],int id)
{
	int i, j;

	system("cls"); //清屏
	printf("\n\t    难度：");
	if (id == 1) 
		printf(" 困难\n");
	if (id == 2) 
		printf(" 一般\n");
	if (id == 3)
		printf(" 简单\n");
	
	//输出上边框
	for (i = 0; i <= SIZE; i++)
	{
		if (i == 0)
		{
			printf("┏");
		}
		else if (i == SIZE)
		{
			printf(" ━━┓");
		}
		else
		{
			printf(" ━");
		}
	}
	printf("\n");
	//输出中间内容
	for (i = 0; i < SIZE; i++)
	{
		for (j = 0; j < SIZE; j++)
		{
			if (j == 0) //输出左边框
			{
				printf("┃ ");
			}
			if (map[i][j] == 1) //输出蛇头
			{
				printf("●");
			}
			else if (map[i][j] > 1) //输出蛇身
			{
				printf("○"); 
			}
			else if (map[i][j] == -1) //输出食物
			{
				printf("★");
			}
			else
			{
				printf("  ");
			}
			if (j == SIZE - 1) //输出右边框
			{
				printf("┃ ");
			}
		}
		printf("\n");
	}
	//输出下边框
	for (i = 0; i <= SIZE; i++) //输出下边框
	{
		if (i == 0)
		{
			printf("┗");
		}
		else if (i == SIZE)
		{
			printf(" ━━┛");
		}
		else
		{
			printf(" ━");
		}
	}
	printf("\n使用“w”“a”“s”“d”控制移动\n");
	printf("Power by Mark0519\n");

}
